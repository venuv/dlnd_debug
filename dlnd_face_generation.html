Preprocess the Data¶

Since the project's main focus is on building the GANs, we'll preprocess the data for you. The values of the MNIST and CelebA dataset will be in the range of -0.5 to 0.5 of 28x28 dimensional images. The CelebA images will be cropped to remove parts of the image that don't include a face, then resized down to 28x28.

The MNIST images are black and white images with a single color channel while the CelebA images have 3 color channels (RGB color channel).

def discriminator(images, reuse=False):
    """
    Create the discriminator network
    :param images: Tensor of input image(s)
    :param reuse: Boolean if the weights should be reused
    :return: Tuple of (tensor output of the discriminator, tensor logits of the discriminator)
    """
    # TODO: Implement Function
    with (tf.variable_scope('discriminator',reuse=reuse)):
        alpha = 0.1
        dropout = 0.5
        
        #kernel filter size 5, 64 filters, stride 2 so dim is halves
        #28x28x1/3
        x1 = tf.layers.conv2d(images, 64, 5, strides=2 ,padding='same')
        leaky_relu1 = tf.maximum(alpha * x1, x1)
        drop1 = tf.nn.dropout(leaky_relu1, dropout)
        #now 14x14x64
        
        x2 = tf.layers.conv2d(drop1, 128, 5, strides=2 ,padding='same')
        bn2 = tf.layers.batch_normalization(x2, training=True)
        leaky_relu2 = tf.maximum(alpha * bn2, bn2)
        drop2 = tf.nn.dropout(leaky_relu2, dropout)
        #now is also 7*7*128
        
        x3 = tf.layers.conv2d(drop2, 256, 5, strides=2 ,padding='same')
        bn3 = tf.layers.batch_normalization(x3, training=True)
        leaky_relu3 = tf.maximum(alpha * bn3, bn3)
        drop3 = tf.nn.dropout(leaky_relu3, dropout)
        #now is 4*4*512
        
        # dense layer setup 
        #flat = tf.reshape(drop3, [-1, 4*4*512])
        #discriminator_logits = tf.layers.dense(flat, 1, kernel_initializer=tf.contrib.layers.xavier_initializer())
        #discriminator_out = tf.nn.sigmoid(discriminator_logits)
        
        disc = tf.reshape(drop3, (-1, 4*4*256))
        discriminator_logits = tf.layers.dense(disc, 1)
        discriminator_out = tf.sigmoid(discriminator_logits)
        
    return discriminator_out,discriminator_logits

Generator¶

Implement generator to generate an image using z. This function should be able to reuse the variables in the neural network. Use tf.variable_scope with a scope name of "generator" to allow the variables to be reused. The function should return the generated 28 x 28 x out_channel_dim images.

def generator(z, out_channel_dim, is_train=True):
    """
    Create the generator network
    :param z: Input z
    :param out_channel_dim: The number of channels in the output image
    :param is_train: Boolean if generator is being used for training
    :return: The tensor output of the generator
    """
    # TODO: Implement Function
    with tf.variable_scope('generator',reuse = not is_train):
        gen = tf.layers.dense(z, 3*3*512)
        gen = tf.reshape(gen, (-1, 3, 3, 512)) # 3x3x512
        gen = tf.layers.batch_normalization(gen, training=is_train)
        gen = tf.maximum(0.1 * gen, gen)
        
        gen = tf.layers.conv2d_transpose(gen, 256, 5, strides=1, padding='valid') # 7x7x256
        gen = tf.layers.batch_normalization(gen, training=is_train)
        gen = tf.maximum(0.1 * gen, gen)
        
        gen = tf.layers.conv2d_transpose(gen, 128, 5, strides=2, padding='same') # 14x14x256
        gen = tf.layers.batch_normalization(gen, training=is_train)
        gen = tf.maximum(0.1 * gen, gen)
        
        logits = tf.layers.conv2d_transpose(gen, out_channel_dim, 5, strides=2, padding='same') # 28x28x[channels]
        out = tf.tanh(logits)
        
        return out

def model_loss(input_real, input_z, out_channel_dim):
    """
    Get the loss for the discriminator and generator
    :param input_real: Images from the real dataset
    :param input_z: Z input
    :param out_channel_dim: The number of channels in the output image
    :return: A tuple of (discriminator loss, generator loss)
    """
    # TODO: Implement Function
    gen_out = generator(input_z, out_channel_dim, True)
    disc_out_real, disc_logits_real = discriminator(input_real, False)
    disc_out_fake, disc_logits_fake = discriminator(gen_out, True)
    
    disc_loss_real = tf.reduce_mean(
        tf.nn.sigmoid_cross_entropy_with_logits(logits=disc_logits_real, 
                                                labels=tf.ones_like(disc_out_real)))
    disc_loss_fake = tf.reduce_mean(
        tf.nn.sigmoid_cross_entropy_with_logits(logits=disc_logits_fake, 
                                               labels=tf.zeros_like(disc_out_fake)))
    gen_loss = tf.reduce_mean(
        tf.nn.sigmoid_cross_entropy_with_logits(logits=disc_logits_fake, 
                                               labels=tf.ones_like(disc_out_fake)))
    disc_loss = disc_loss_real + disc_loss_fake
    
    return disc_loss, gen_loss
    #return None, None

def model_opt(d_loss, g_loss, learning_rate, beta1):
    """
    Get optimization operations
    :param d_loss: Discriminator loss Tensor
    :param g_loss: Generator loss Tensor
    :param learning_rate: Learning Rate Placeholder
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :return: A tuple of (discriminator training operation, generator training operation)
    """
    # TODO: Implement Function
    all_vars = tf.trainable_variables()
    disc_vars = [var for var in all_vars if var.name.startswith('discriminator')]
    gen_vars = [var for var in all_vars if var.name.startswith('generator')]
    
    with tf.control_dependencies(tf.get_collection(tf.GraphKeys.UPDATE_OPS)):
        disc_train_opt = tf.train.AdamOptimizer(learning_rate, beta1=beta1).minimize(d_loss, var_list=disc_vars)
        gen_train_opt = tf.train.AdamOptimizer(learning_rate, beta1=beta1).minimize(g_loss, var_list=gen_vars)
    
    return disc_train_opt, gen_train_opt
    #return None, None

def train(epoch_count, batch_size, z_dim, learning_rate, beta1, get_batches, data_shape, data_image_mode):
    """
    Train the GAN
    :param epoch_count: Number of epochs
    :param batch_size: Batch Size
    :param z_dim: Z dimension
    :param learning_rate: Learning Rate
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :param get_batches: Function to get batches
    :param data_shape: Shape of the data
    :param data_image_mode: The image mode to use for images ("RGB" or "L")
    """
    # TODO: Build Model
    class GAN:
        def __init__(self, real_size, z_size, learning_rate, beta1):
            # tf.reset_default_graph()
            
            self.input_real, self.input_z, self.learning_rate = model_inputs(
                                                    real_size[1], real_size[2], real_size[3], z_size)
            
            self.learning_rate = learning_rate
            
            self.disc_loss, self.gen_loss = model_loss(self.input_real, self.input_z, real_size[3])
            
            self.disc_opt, self.gen_opt = model_opt(self.disc_loss, self.gen_loss, self.learning_rate, beta1)
            
    net = GAN(data_shape, z_dim, learning_rate, beta1)
    
    saver = tf.train.Saver()
    !mkdir checkpoints
    steps = 0
    
    with tf.Session() as sess:
        sess.run(tf.global_variables_initializer())
        for epoch_i in range(epoch_count):
            for batch_images in get_batches(batch_size):
                # TODO: Train Model
                steps += 1
                
                batch_z = np.random.uniform(-1, 1, size=(batch_size, z_dim))
                
                _ = sess.run(net.disc_opt, feed_dict={net.input_real: batch_images, net.input_z: batch_z})
                _ = sess.run(net.gen_opt, feed_dict={net.input_z: batch_z, net.input_real: batch_images})
                
                # Print progress
                if steps % 10 == 0:
                    train_loss_disc = net.disc_loss.eval({net.input_z: batch_z, net.input_real: batch_images})
                    train_loss_gen = net.gen_loss.eval({net.input_z: batch_z})
                    
                    print("Epoch {}/{}...".format(epoch_i+1, epoch_count), 
                         "Step {}...".format(steps), 
                         "Discriminator Loss: {:.4f}...".format(train_loss_disc), 
                         "Generator Loss: {:.4f}".format(train_loss_gen))
                
                # Show samples
                if steps % 100 == 0:
                    show_generator_output(sess, 25, net.input_z, data_shape[3], data_image_mode)
                    
        saver.save(sess, './checkpoints/generator.ckpt')
        show_generator_output(sess, 25, net.input_z, data_shape[3], data_image_mode)
